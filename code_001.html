統計学をやってみた

分散と標準偏差

#include 
#include 

using namespace std;

int main(void)
{
	int array[] = {51, 49, 50, 57, 43};
	int element_num = sizeof(array) / sizeof(array[0]);
	int sum = 0;

	for(int i=0;i<element_num;i++)
	{
		sum += array[i];
	}

	int average = sum / element_num;
	sum = 0;

	for(int i=0;i<element_num;i++)
	{
		int tmp = (array[i] - average);
		sum += tmp * tmp;
	}

	int dispersion = sum / element_num;
	float standard_diviation = sqrt(dispersion);

	cout << dispersion << endl;
	cout << standard_diviation << endl;

	return 0;
}
